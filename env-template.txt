# Examples: nyc3.digitaloceanspaces.com, sfo3.digitaloceanspaces.com, sgp1.digitaloceanspaces.com

# Your Space region (must match the endpoint)
# Examples: nyc3, sfo3, sgp1, fra1, ams3, lon1

# Your Space name (bucket name)
# Find this in: https://cloud.digitalocean.com/spaces

# Your Spaces Access Key
# Find/create in: https://cloud.digitalocean.com/account/api/spaces
# Starts with DO00...

# Your Spaces Secret Key
# Generated when you create the Access Key (only shown once!)
SPACES_SECRET_KEY=your-secret-key-here

# Your CDN URL
# Format: https://[bucket-name].[region].cdn.digitaloceanspaces.com
# Find in your Space settings
